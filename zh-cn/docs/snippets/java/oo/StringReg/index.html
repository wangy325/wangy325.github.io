    Java正则匹配字符串小例

    2021-02-07
    Java
    Snippet
    2025-04-02
     1/**
 2 * @author wangy
 3 * @date 2021-2-7 13:39
 4 */
 5
 6public class StringReg {
 7    static String pattern1 = "ab.cd";
 8    static String pattern2 = "ab.cd.ef";
 9    static String pattern3 = "java.util.String";
10    static String pattern4 = "ok";
11    static String pattern5 =
12        "/profile/upload/2020/09/15/9d6ca80b4030ec3589.png";
13
14    static String pattern6 = "hello@mary";
15
16    /** 判断字符串是否匹配 */
17    static String pattern12(String reg) {
18        boolean b1 = Pattern.matches(reg, pattern1);
19        boolean b2 = Pattern.matches(reg, pattern2);
20        boolean b3 = Pattern.matches(reg, pattern3);
21        boolean b4 = Pattern.matches(reg, pattern4);
22        return b1 + ", " + b2 + ", " + b3 + ", " + b4;
23    }
24
25    /** 匹配并替换首字母 */
26    static String replaceFirst(String reg) {
27//        return pattern1
28//          .replaceFirst(reg, String.valueOf(pattern1.charAt(0))
29//               .toUpperCase());
30        return Pattern
31            .compile(reg)
32            .matcher(pattern1)
33            .replaceFirst(pattern1.substring(0, 1).toUpperCase());
34    }
35
36    /** 获取文件的格式后缀 */
37    static String getSuffix(String reg, String target) {
38        Matcher matcher = Pattern.compile(reg).matcher(target);
39        if (matcher.find()) {
40            return matcher.group(1);
41        }
42        return null;
43    }
44
45    public static void main(String[] args) {
46        System.out.println(pattern12("^([a-z]+\\.)+[a-z]+$"));
47        System.out.println(replaceFirst("\\w"));
48        System.out.println(getSuffix("(\\.\\w{3,4})", pattern5));
49        System.out.println(getSuffix("@(\\S+)", pattern6));
50    }
51}
