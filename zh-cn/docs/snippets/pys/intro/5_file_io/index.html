      1"""
  2Author: wangy325
  3Date: 2024-08-06 11:01:31
  4Description: 
  5"""
  6from math import pi as pi
  7import os
  8
  9"""
 10输入输出与文件I/O
 111. 格式化字符串
 122. 读取文件
 13"""
 14'''
 15使用占位符
 16'''
 17year = 2024
 18event = 'Trump\'s Gun Shot'
 19print(f'Big News of {event} in {year}')
 20
 21# 或者这样
 22print(f'Big News of {event=} in {year=}')
 23
 24# 占位符支持还格式化字符串输出
 25print(f'The value of pi is approximately {pi:.3f}')
 26
 27# 可以指定宽度
 28table = {'Sjoerd': 4127, 'Jack': 4098, 'Dcab': 7678}
 29for name, phone in table.items():
 30    print(f'{name:10} ==> {phone:10d}')
 31'''
 32使用format()方法
 33'''
 34yes_votes = 42_573_651
 35total_votes = 83_637_912
 36percentage = yes_votes / total_votes
 37print('{:-9} YES votes {:2.3%}'.format(yes_votes, percentage))
 38
 39# 或者 加入前导位置索引
 40print('{1:-9} YES votes {0:2.3%}'.format(percentage, yes_votes))
 41
 42# 或者直接使用关键字
 43print('{yes_votes:4} YES votes {percentage:2.3%}'.format(yes_votes=117,
 44                                                         percentage=0.459417))
 45# 还可以直接输出字典类型
 46dicta = {'Sjoerd': 4127, 'Jack': 4098, 'Dcab': 8637678}
 47print('Jack: {Jack:d}; Sjoerd: {Sjoerd:d}; Dcab: {Dcab:d}'.format(**dicta))
 48'''
 49Old school 格式化方法(Java还在用哦)
 50'''
 51print('The value of pi is approximately %5.3f.' % pi)
 52
 53'''
 54读写文件
 55python可以以纯文本或者二进制字节的形式读写文件
 56with 关键字的作用类似Java的try..with...resource, 不需要显式地处理异常和关闭资源
 57如果不使用with关键字, 则需要处理异常 和关闭资源
 58
 59with语句在python中更加强大, 一般称作"上下文管理器"
 60https://docs.python.org/zh-cn/3/reference/compound_stmts.html#the-with-statement
 61'''
 62# ###### #
 63# 文本流  #
 64# ###### #
 65with open('pys/misc/text', '+w', encoding="utf-8") as f:
 66    f.writelines([
 67        '這是一個純文字檔案.\n',
 68        'This is a plain text file.\n'
 69        'これはただのテキストファイルです.\n',
 70        'Il s\'agit d\'un fichier texte brut.\n',
 71        'Este es un archivo de texto sin formato.\n'
 72    ])
 73fil = open('pys/misc/text')
 74while (line := fil.readline()) != '':
 75    print(line, end='')
 76fil.close()
 77
 78# ###### #
 79# 字节流  #
 80# ###### #
 81
 82file_path = 'pys/misc/textb'
 83if not os.path.exists(file_path) or not os.path.isfile(file_path):
 84    open('pys/misc/textb', 'x')
 85with open('pys/misc/textb', '+wb') as fb:
 86    btsw = fb.write(b'This is a binary text file.')
 87    # 文件指针向前移动到写入字节数前一位
 88    fb.seek(btsw - 1)
 89    bt = fb.read(1)
 90    print(bt == b'.')
 91
 92
 93# 试试复制张图片
 94def copy_file(target, dest):
 95    if os.path.exists(target) and os.path.isfile(target):
 96        if not os.path.exists(dest) or not os.path.isfile(dest):
 97            df = open(dest, '+xb')
 98            with open(target, 'rb') as tf:
 99                while len((buffer := tf.read(1024))) > 0:
100                    df.write(buffer)
101            df.close()
102        else:
103            print("dest file exists.")
104    else:
105        raise FileNotFoundError("target file not found.")
106
